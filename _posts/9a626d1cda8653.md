  collection,
  addDoc,
  onSnapshot, // ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö real-time updates
  query,
  orderBy,
  serverTimestamp, // ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö timestamp ‡∏Ç‡∏≠‡∏á server
  doc,
  updateDoc,
  deleteDoc,

// Import Components

import TodoList from './components/TodoList';

function App() {

  const [loading, setLoading] = useState(true); // State ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö Loading

  useEffect(() => {

    const q = query(collection(db, 'todos'), orderBy('createdAt', 'desc')); // ‡πÄ‡∏£‡∏µ‡∏¢‡∏á‡∏•‡πà‡∏≤‡∏™‡∏∏‡∏î‡∏Ç‡∏∂‡πâ‡∏ô‡∏Å‡πà‡∏≠‡∏ô

    // onSnapshot ‡∏à‡∏∞‡∏ó‡∏≥‡∏á‡∏≤‡∏ô‡∏ó‡∏∏‡∏Å‡∏Ñ‡∏£‡∏±‡πâ‡∏á‡∏ó‡∏µ‡πà‡∏°‡∏µ‡∏Å‡∏≤‡∏£‡πÄ‡∏õ‡∏•‡∏µ‡πà‡∏¢‡∏ô‡πÅ‡∏õ‡∏•‡∏á‡πÉ‡∏ô collection

    }, (error) => {

    return () => unsubscribe();
  }, []); // useEffect ‡∏ô‡∏µ‡πâ‡∏à‡∏∞‡∏ó‡∏≥‡∏á‡∏≤‡∏ô‡∏Ñ‡∏£‡∏±‡πâ‡∏á‡πÄ‡∏î‡∏µ‡∏¢‡∏ß‡∏ï‡∏≠‡∏ô component mount

        text: text,
        completed: false,
        createdAt: serverTimestamp(), // ‡πÉ‡∏ä‡πâ timestamp ‡∏à‡∏≤‡∏Å server

    } catch (error) {
      console.error("Error adding todo: ", error);
      // ‡∏Ñ‡∏ß‡∏£‡∏°‡∏µ feedback ‡πÉ‡∏´‡πâ user ‡∏ó‡∏£‡∏≤‡∏ö‡∏ß‡πà‡∏≤‡πÄ‡∏Å‡∏¥‡∏î‡∏Ç‡πâ‡∏≠‡∏ú‡∏¥‡∏î‡∏û‡∏•‡∏≤‡∏î
    }
  };

  // 3. Update (Toggle Complete Status)
  const handleToggleComplete = async (id: string, completed: boolean) => {
    try {
      const todoRef = doc(db, 'todos', id); // ‡∏≠‡πâ‡∏≤‡∏á‡∏≠‡∏¥‡∏á document ‡∏î‡πâ‡∏ß‡∏¢ id
      await updateDoc(todoRef, {
        completed: completed, // ‡∏≠‡∏±‡∏õ‡πÄ‡∏î‡∏ï field 'completed'
      });
    } catch (error) {
      console.error("Error updating todo: ", error);
    }
  };

  // 4. Delete (Remove Todo)
  const handleDeleteTodo = async (id: string) => {
    if (window.confirm("‡∏ï‡πâ‡∏≠‡∏á‡∏Å‡∏≤‡∏£‡∏•‡∏ö‡∏£‡∏≤‡∏¢‡∏Å‡∏≤‡∏£‡∏ô‡∏µ‡πâ‡πÉ‡∏ä‡πà‡∏´‡∏£‡∏∑‡∏≠‡πÑ‡∏°‡πà?")) { // ‡∏¢‡∏∑‡∏ô‡∏¢‡∏±‡∏ô‡∏Å‡πà‡∏≠‡∏ô‡∏•‡∏ö
        try {
            const todoRef = doc(db, 'todos', id);
            await deleteDoc(todoRef);
        } catch (error) {
            console.error("Error deleting todo: ", error);
        }
    }
  };

  // --- Render UI ---
  return (
    <div className="container mx-auto p-4 max-w-lg">
      <div className="card bg-base-100 shadow-xl">
        <div className="card-body">
          <h1 className="card-title text-3xl justify-center mb-4">
             Todo List (React + TS + Firestore + DaisyUI)
          </h1>

          {/* Form ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö‡πÄ‡∏û‡∏¥‡πà‡∏° Todo */}
          <AddTodoForm onAddTodo={handleAddTodo} />

          {/* ‡πÅ‡∏™‡∏î‡∏á Loading ‡∏´‡∏£‡∏∑‡∏≠ ‡∏£‡∏≤‡∏¢‡∏Å‡∏≤‡∏£ Todo */}
          {loading ? (
             <div className="text-center">
                <span className="loading loading-spinner loading-lg text-primary"></span>
                <p>‡∏Å‡∏≥‡∏•‡∏±‡∏á‡πÇ‡∏´‡∏•‡∏î‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•...</p>
             </div>
          ) : (
            <TodoList
              todos={todos}
              onToggleComplete={handleToggleComplete}
              onDeleteTodo={handleDeleteTodo}
            />
          )}
